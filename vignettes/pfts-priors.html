Bayesian Model Parameterization using betydata

Introduction

This vignette demonstrates how to work with Plant Functional Types (PFTs) and Bayesian prior distributions using betydata. These are central to the PEcAn ecosystem modeling workflow.

Key concepts:

  • PFT (Plant Functional Type): A grouping of species with similar functional characteristics used for ecosystem modeling
  • Prior: A probability distribution encoding expert knowledge about a parameter before observing data
  • Posterior: Updated distribution after combining prior with observations

Visualizing a Prior Distribution

# Get SLA priors first
sla_var_id <- variables |> 
  filter(name == "SLA") |> 
  pull(id)

sla_priors <- priors |>
  filter(variable_id %in% sla_var_id)

# Function to evaluate any prior distribution
eval_prior_density <- function(p, x = NULL) {
  # Generate appropriate x range if not provided
  if (is.null(x)) {
    x <- switch(p$distn,
      "gamma" = seq(0, qgamma(0.999, p$parama, p$paramb), length.out = 1000),
      "norm" = seq(p$parama - 4*p$paramb, p$parama + 4*p$paramb, length.out = 1000),
      "lnorm" = seq(0.001, qlnorm(0.999, p$parama, p$paramb), length.out = 1000),
      "beta" = seq(0.001, 0.999, length.out = 1000),
      "unif" = seq(p$parama, p$paramb, length.out = 1000),
      "weibull" = seq(0, qweibull(0.999, p$parama, p$paramb), length.out = 1000),
      "exp" = seq(0, qexp(0.999, p$parama), length.out = 1000),
      seq(0, 100, length.out = 1000)
    )
  }
  
  y <- switch(p$distn,
    "gamma"   = dgamma(x, shape = p$parama, rate = p$paramb),
    "norm"    = dnorm(x, mean = p$parama, sd = p$paramb),
    "lnorm"   = dlnorm(x, meanlog = p$parama, sdlog = p$paramb),
    "beta"    = dbeta(x, shape1 = p$parama, shape2 = p$paramb),
    "unif"    = dunif(x, min = p$parama, max = p$paramb),
    "weibull" = dweibull(x, shape = p$parama, scale = p$paramb),
    "exp"     = dexp(x, rate = p$parama),
    rep(0, length(x))  # fallback
  )
  
  list(x = x, y = y)
}

# Plot if SLA priors exist
if (nrow(sla_priors) > 0) {
  p <- sla_priors[1, ]
  result <- eval_prior_density(p)
  
  ggplot(data.frame(x = result$x, y = result$y), aes(x, y)) +
    geom_line(linewidth = 1) +
    labs(
      title = "Prior distribution for SLA",
      subtitle = paste0(p$distn, "(", round(p$parama, 2), ", ", round(p$paramb, 2), ")"),
      x = "SLA (m2/kg)",
      y = "Density"
    ) +
    theme_minimal()
}

Extracting Data for a PFT

A common workflow: get all trait data for species in a given PFT.

# Function to get trait data for a PFT
get_pft_data <- function(pft_name, trait_name = NULL) {
  # Find PFT id
  pft_id <- pfts |> 
    filter(grepl(pft_name, name, ignore.case = TRUE)) |>
    pull(id)
  
  if (length(pft_id) == 0) {
    warning("PFT not found: ", pft_name)
    return(NULL)
  }
  
  # Get species in this PFT
  spp_ids <- pfts_species |>
    filter(pft_id %in% !!pft_id) |>
    pull(specie_id)
  
  # Filter traitsview
  result <- traitsview |>
    filter(species_id %in% spp_ids)
  
  if (!is.null(trait_name)) {
    result <- result |> filter(trait == trait_name)
  }
  
  result
}
